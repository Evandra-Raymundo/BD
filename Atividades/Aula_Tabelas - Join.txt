SELECT A.Nome, B.Nome
FROM TabelaA as A
LEFT JOIN TabelaB as B on A.Nome = B.Nome;

SELECT A.Nome, B.Nome
FROM TabelaA as A RIGHT JOIN TabelaB as B on A.Nome = B.Nome;
