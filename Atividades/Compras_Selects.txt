18)Selecione nome_cliente e endereço da tabela de  cliente e também o num_pedido e o prazo_entrega da tabela de pedidos onde o código 

SELECT  c.nome_cliente,c.endereco,p.num_pedido,p.prazo_entrega

19)Selecione o codigo_vendedor e o nome_vendedor da tabela de vendedor e num_pedido e o prazo_entrega da tabela de pedidos para prazos de entrega  iguais a 20.

SELECT v.codigo_vendedor,v.nome_vendedor,p.num_pedido,p.prazo_entrega
FROM vendedor v INNER JOIN pedido p
ON v.codigo_vendedor = p.codigo_vendedor
Where p.prazo_entrega =20;


20) Selecione codigo_produto e a descricao_produto da tabela de produto e a quantidade da tabela item_pedido onde a quantidade seja menor ou igual a 18

SELECT p.codigo_produto,p.descricao_produto,i.quantidade
FROM produto p  INNER JOIN item_pedido i
ON  p.codigo_produto = i.codigo_produto
where i.quantidade <=18


21) Selecione codigo_produto,descricao da tabela produto e quantidade da tabela item_produto onde a descrição comece com a letra "C" 

SELECT p.codigo_produto, p.descricao_produto, i.quantidade 
FROM produto p
INNER JOIN  item_pedido i ON p.codigo_produto = i.codigo_produto
where p.descricao_produto like"C%"


22) Selecione o nome do cliente da tabela cliente, nome do vendedor da tabela vendedor, número do pedido, prazo de entrega da tabela pedido, 
código do produto e descrição do produto da tabela de produto e quantidade da tabela item do produto onde  a quantidade seja maior que 15

SELECT c.nome_cliente, v.nome_vendedor, pd.num_pedido,pd.prazo_entrega,p.codigo_produto,p.descricao_produto, i.quantidade
FROM  produto p
inner join  item_pedido i
  on p.codigo_produto = i.codigo_produto
inner join  pedido pd
  on pd.num_pedido = i.num_pedido
  inner join cliente c
   on c.codigo_cliente = pd.codigo_cliente
inner join vendedor v
  on v.codigo_vendedor = pd.codigo_vendedor
where i.quantidade>15
